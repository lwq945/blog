## 利用对象的键名不能重复特性
```
function unique(array) {
  let newArr = [];
  let obj = {};
  for(let i =0; i < array.length; i++) {
    obj[array[i]] = array[i];
  }
  
  for(let key in obj) {
    newArr.push(obj[key])
  }
  return newArr;
}
```

## 使用数组indexOf()方法
```
function unique(array) {
  let newArr = [];
  for(let i =0; i < array.length; i++) {
    if(newArr.indexOf(array[i]) === -1){
      newArr.push(array[i])
    }
  }
  return newArr;
}
```
## 使用ES6的Set数据结构
```
function unique(array) {
  return Array.from(new Set(array))
}

//或者扩展运算符(...)
function unique(array) {
  return [...new Set(array)]
}

```
